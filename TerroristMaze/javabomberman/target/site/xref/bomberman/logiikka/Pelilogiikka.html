56          for (int i = 1; i <= 2; i++) {
57              Pelaaja p = this.inventaario.getPelaaja(i);
58              if (p.getJattaakoPommin()) {

65      /**
66       * Metodi edistaa kaikkien Pommi olioiden ajastinta ja luo uuden rajahdyksen
67       * jos pommin ajastin on 0. Pommeja joiden ajastin on 0 ei lisata listaan
68       * joka korvaa vanhan pommilistan
69       *
70       */
71      public void etenePommit() {
72          ArrayList<Pommi> uusiPommilista = new ArrayList<Pommi>();
73          for (Pommi pommi : this.inventaario.getPommit()) {
74              pommi.etene();
75              if (pommi.getTimer() <= 0) {
76                  this.inventaario.getPelaaja(pommi.getOmistaja()).lisaaKaytettavaPommi();
77                  luoRajahdykset(pommi.getX(), pommi.getY(), pommi.getVoima());
78              } else {
79                  uusiPommilista.add(pommi);
80              }
81          }
82          this.inventaario.setPommilista(uusiPommilista);
83      }

134     private void tapaPelaajatRuudusta(int x, int y) {
135         for (int i = 1; i <= 2; i++) {
136             Pelaaja p = this.inventaario.getPelaaja(i);
137             int pX = p.getX() + 15;
138             int pY = p.getY() + 15;
139             int rX = x * 50 + 25;
140             int rY = y * 50 + 25;
141             // pelaajan ja rajahdyksen keskipisteet
142             if (tsebysovEtaisyys(etaisyysItseisArvo(pX, rX), etaisyysItseisArvo(pY, rY)) < 40) {
143                 p.setHavisi();
144             }
145         }
146     }

148     private int tsebysovEtaisyys(int x, int y) {
149         if (x > y) {
150             return x;
151         }
152         return y;
153     }

155     private int etaisyysItseisArvo(int x1, int x2) {
156         if (x1 - x2 > 0) {
157             return x1 - x2;
158         }
159         return x2 - x1;
160     }

209      * Metodia kaytetaan peli olion luonnin yhteydessa varmistamaan peliin
210      * liittyvien seinien olemassa olo ja sijainti.
211      *
212      */
213     public void luoSeinat() {
214         ArrayList<Seina> seinat = new ArrayList<Seina>();
215         for (int i = 1; i < 12; i++) {
216             for (int j = 1; j < 12; j++) {
217                 if (i + j > 3 && i + j < 21) {
218                     if (i % 2 == 1 && j % 2 == 0) {
219                         seinat.add(new Seina(i, j));
220                     }
221                     if (i % 2 == 0 && j % 2 == 1) {
222                         seinat.add(new Seina(i, j));
223                     }
224                 }
225             }
226         }
227         this.inventaario.setUusiSeinaLista(seinat);
228     }

230     /**
231      * Metodia rakentaa pohjan ja EsineInventaarion Seina-listan avulla ruudukon
232      * jota kaytetaan javabomberman liikkumisen pohjana
233      *
234      * @return pelilaudan rakenne 13x13 ruudukossa
235      */
236     public Ruutu[][] luoPieniRuudukko() {
237         // 0 = vapaa, 1 = kiveÃ¤, 2 = mutaseina
238         Ruutu[][] vapaana = new Ruutu[13][13];
239         for (int i = 0; i < 13; i++) {
240             for (int j = 0; j < 13; j++) {
241                 if (i == 0 || i == 12 || j == 0 || j == 12) {
242                     vapaana[i][j] = Ruutu.KIVISEINA;
243                 } else if (i % 2 == 0 && j % 2 == 0) {
244                     vapaana[i][j] = Ruutu.KIVISEINA;
245                 } else {
246                     vapaana[i][j] = Ruutu.VAPAA;
247                 }
248             }
249         }
250 
251         for (Seina seina : this.inventaario.getSeinat()) {
252             vapaana[seina.getX()][seina.getY()] = Ruutu.MUTASEINA;
253         }
254         for (Pommi pommi : this.inventaario.getPommit()) {
255             vapaana[pommi.getX()][pommi.getY()] = Ruutu.POMMI;
256         }
257         vapaana[this.inventaario.getPelaaja(1).getX() / 50][this.inventaario.getPelaaja(1).getY() / 50] = Ruutu.PELAAJA;
258         vapaana[this.inventaario.getPelaaja(2).getX() / 50][this.inventaario.getPelaaja(2).getY() / 50] = Ruutu.PELAAJA;
259         return vapaana;
260     }

263      * Metodi luo uudet rajahdykset haluttuun ruutuun ja sen viereisiin
264      * ruutuihin, riippuen asetetusta voima parametrista. Metodi kutsuu
265      * luoYksiRajahdysSuunta -Metodia nelja kertaa, kerran jokaista ilmansuuntaa
266      * kohti.
267      *
268      * @param x x -koordinaatti
269      * @param y y -koordinaatti
270      * @param voima Rajahdyksen pituus
271      *
272      * @see bomberman.object.Rajahdys
273      */
274     public void luoRajahdykset(int x, int y, int voima) {
275         Ruutu[][] ruudukko = luoPieniRuudukko();
276         this.inventaario.luoRajahdys(x, y);
277         luoYksiRajahdysSuunta(1, 0, x, y, voima, ruudukko);
278         luoYksiRajahdysSuunta(-1, 0, x, y, voima, ruudukko);
279         luoYksiRajahdysSuunta(0, 1, x, y, voima, ruudukko);
280         luoYksiRajahdysSuunta(0, -1, x, y, voima, ruudukko);
281     }

283     /**
284      * Metodi luo uusia Rajahdys-olioita. Annetun pelitilanteen mukaan uusien
285      * rajahdysten luonti saataa pysahtya. Metodi toimii iteratiivisesti.
286      *
287      * @param dx suunta x -akselilla
288      * @param dy suunta y -akselilla
289      * @param x x -koordinaatti
290      * @param y y -koordinaatti
291      * @param voima Rajahdyksen pituus
292      * @param ruudukko annettu pelitilanne
293      *
294      */
295     public void luoYksiRajahdysSuunta(int dx, int dy, int x, int y, int voima, Ruutu[][] ruudukko) {
296         while (voima > 0) {
297             x = x + dx;
298             y = y + dy;
299             if (ruudukko[x][y] != Ruutu.KIVISEINA) {
300                 this.inventaario.luoRajahdys(x, y);
301                 if (ruudukko[x][y] == Ruutu.MUTASEINA) {
302                     voima = 0;
303                 }
304             } else {
305                 voima = 0;
306             }
307             voima = voima - 1;
308         }
309     }

346      * Metodi siirtaa EsineInventaario olioon sailottyja pelaajia
347      * tarkistaSijainninLaillisuus avulla, jonka jÃ¤lkeen se tarkistaa saiko
348      * pelaaja PowerUpia
349      *
350      * bomberman.logiikka.SeinaType[][])
351      */
352     private void etenePelaaja(Pelaaja p, Ruutu[][] r) {
353         int nopeus = p.getNopeus();
354         while (nopeus > 0) {
355             if (sijainti(r, p, p.getNopeusX(), 0)) {
356                 p.siirra(p.getNopeusX(), 0);
357             }
358             if (sijainti(r, p, 0, p.getNopeusY())) {
359                 p.siirra(0, p.getNopeusY());
360             }
361             nopeus--;
362         }
363         tarkistaPowerUp(p);
364     }

375     /**
376      * Metodi tutkii onko annetulle pelaajalle sallittua liikkua 1 askel
377      * eteenpain.
378      *
379      * @param r Ruutu[][] koordinaatisto
380      * @param p annettu Pelaaja olio
381      * @param dX pelaajan nopeus X suunnassa
382      * @param dY pelaajan nopeus Y suunnassa
383      * @return
384      */
385     public boolean sijainti(Ruutu[][] r, Pelaaja p, int dX, int dY) {
386         int x = p.getX() + dX;
387         int y = p.getY() + dY;
388         //Metodi etsii tilanteet joissa sijainti ei kelpaa
389         if (r[x / 50][y / 50] != Ruutu.PELAAJA) {
390             if (r[x / 50][y / 50] != Ruutu.VAPAA) {
391                 if(x%50 > 40 || y%50 > 40){
392                 return false; 
393                 }
394                 // jos sijainnissa ei ole pelaaja tai se ei ole vapaa niin sijainti on laiton
395             }
396         }
397         if (x % 50 > 20 && y % 50 > 20) {
398             return false;
399             // jos sijainti on oikeassa alareunassa sita ei hyvaksyta
400         }
401         if (y % 50 <= 20 && x % 50 > 20) {
402             if ((r[(x / 50) + 1][y / 50] == Ruutu.KIVISEINA || r[(x / 50) + 1][y / 50] == Ruutu.MUTASEINA)) {
403                 return false;
404                 //seinan sisaan meneminen ei ole sallittu
405             }
406             if (r[(x / 50) + 1][y / 50] == Ruutu.POMMI) {
407                 if(p.getNopeusX() > 0 && p.getX()%50 <= 20){
408                     return false;
409                 }
410                 //pommia kohti ei saa menna
411             }
412         }
413         if (y % 50 > 20 && x % 50 <= 20) {
414             if ((r[x / 50][(y / 50) + 1] == Ruutu.KIVISEINA || r[x / 50][(y / 50) + 1] == Ruutu.MUTASEINA)) {
415                 return false;
416                 //seinan sisaan meneminen ei ole sallittu
417             }
418             if (r[x / 50][(y / 50) + 1] == Ruutu.POMMI) {
419                 if(p.getNopeusY() > 0 && p.getY()%50 <= 20){
420                     return false;
421                 }
422                 
423                 //pommia kohti ei saa menna
424             }
425         }
426         return true;
427     }
