6   package bomberman.object;
7   
8   /**
9    * Luokka kuvaa Seina oliota
10   *
11   * @author jaollika@cs
12   */
13  public class Seina {
14  
15      private int sijaintiX;
16      private int sijaintiY;
17      private boolean tuhoutumassa;
18      private int vauriopisteet;
19  
20      public Seina(int x, int y) {
21          this.sijaintiX = x;
22          this.sijaintiY = y;
23          this.tuhoutumassa = false;
24          this.vauriopisteet = 8;
25      }
26  
27      public int getX() {
28          return this.sijaintiX;
29      }
30  
31      public int getY() {
32          return this.sijaintiY;
33      }
34  
35      public void setTuhoutuu() {
36          this.tuhoutumassa = true;
37      }
38  
39      public boolean getTuhoutuu() {
40          return this.tuhoutumassa;
41      }
42  
43      /**
44       * Metodi vahentaa Seina olion vauriopisteet arvoa ja kutsuu setTuhoutuu()
45       * jos seinan vauriopisteet laskevat nollaan tai sen alle.
46       */
47      public void otaVauriota() {
48          this.vauriopisteet--;
49          if (this.vauriopisteet <= 0) {
50              this.setTuhoutuu();
51          }
52      }
53  
54  }
