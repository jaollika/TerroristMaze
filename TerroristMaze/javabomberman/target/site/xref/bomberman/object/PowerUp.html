6   package bomberman.object;
7   
8   import bomberman.Enum.PowerUpTyyppi;
9   import java.util.Random;
10  
11  /**
12   * Luokka kuvaa pelissa kaytettavaa PowerUp oliota.
13   *
14   * @author -
15   */
16  public class PowerUp {
17  
18      private PowerUpTyyppi tyyppi;
19      private int x;
20      private int y;
21  
22      /**
23       * Metodi luo uuden tietylla PowerUpType arvolla varustetun PowerUp olion.
24       *
25       * @param x koordinaatti
26       * @param y koordinaatti
27       * @param tyyppi PowerUpin tyyppi
28       */
29      public PowerUp(int x, int y, PowerUpTyyppi tyyppi) {
30          this.x = x;
31          this.y = y;
32          this.tyyppi = tyyppi;
33      }
34  
35      /**
36       * Metodi luo uuden satunnaisella PowerUpType arvolla varustetun PowerUp
37       * olion.
38       *
39       * @param x
40       * @param y
41       */
42      public PowerUp(int x, int y) {
43          this.x = x;
44          this.y = y;
45          Random random = new Random();
46          int i = random.nextInt(9);
47          if (i < 4) {
48              this.tyyppi = PowerUpTyyppi.VOIMA;
49          } else if (i < 7) {
50              this.tyyppi = PowerUpTyyppi.NOPEUS;
51          } else {
52              this.tyyppi = PowerUpTyyppi.POMMEJA;
53          }
54      }
55  
56      public int getX() {
57          return this.x;
58      }
59  
60      public int getY() {
61          return this.y;
62      }
63  
64      public PowerUpTyyppi getType() {
65          return this.tyyppi;
66      }
67  }
