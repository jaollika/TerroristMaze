19  /**
20   * Luokka huolehtii että peli etenee askel askeleelta oikein
21   * @author jaollika@cs
22   */
23  public class PeliMoottori{
24  
25      /**
26       * PeliMoottorin kayttama peli-olio
27       */
28      public Pelilogiikka peli;
29  
30      /**
31       * Metodi aloittaa bomberman pelin pyörityksen
32       * @throws InterruptedException
33       */
34      public void start() throws InterruptedException {
35          peli = new Pelilogiikka();
36          Tulos pyorii = Tulos.NYD;
37          long aika = System.nanoTime();
38          while (pyorii == Tulos.NYD) {
39              if(System.nanoTime() - aika > 100000000){
40                  paivitaPeli(peli);
41                  pyorii = peli.tarkistaVoittaja();
42                  aika = System.nanoTime();
43                  peli.paivitagrafiikka();
44                  Thread.sleep(1);
45              }
46              peli.paivitagrafiikka();
47              Thread.sleep(10);
48          }
49          Thread.sleep(5000);
50          peli.sammuta(pyorii);
51  
52      }
53  
54      /**
55       * Metodi paivittaa jarjestyksessa pelilogiikan osat
56       *
57       * @param peli
58       */
59      public void paivitaPeli(Pelilogiikka peli) {
60          peli.etenePelaajat();
61          peli.jataPommit();
62          peli.etenePommit();
63          peli.eteneRajahdykset();
64          peli.paivitaSeinat();
65      }
